create database Hotel;
use Hotel;

create table Habitaciones(
ID_habitacion int(3) auto_increment primary key,
tipo varchar (40),
precio int (5),
capacidad int (1),
estado varchar (70)
 );
 
 create table Reservas(
 ID_reserva int(4) auto_increment primary key,
 fecha_entrada date,
 fecha_salida date,
 foreign key (clientes) references clientes,
 foreign key (habitaciones) references habitaciones
 );
 
 create table Servicios(
 ID_servicio int(2) auto_increment primary key,
 nombre varchar (20),
 precio int(10),
 foreign key (reserva) references reservas
 );
 
 create table Clientes(
 ID_cliente int(4) auto_increment primary key,
 nombre varchar(30),
 dni int(8),
 telefono int(10)
 );
 
 create table Historial_Reservas(
 ID_historial int(4) auto_increment primary key,
 cliente varchar (40),
 reserva datetime,
 foreign key (reserva) references reserva
 );
