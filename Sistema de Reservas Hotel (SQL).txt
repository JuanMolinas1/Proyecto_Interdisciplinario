create database Hotel;
use Hotel;

create table Habitaciones(
ID_habitacion INT NOT NULL primary key,
tipo varchar (40),
precio int (5),
capacidad int (1),
estado varchar (70)
 );
 
 create table Reservas(
 ID_reserva int NOT NULL primary key,
 fecha_entrada date,
 fecha_salida date,
 foreign key (clientes) references clientes,
 foreign key (habitaciones) references habitaciones
 );
 
 create table Servicios(
 ID_servicio INT NOT NULL primary key,
 nombre varchar (20),
 precio int(10),
 foreign key (reserva) references reservas
 );
 
 create table Clientes(
 ID_cliente INT NOT NULL primary key,
 nombre varchar (30),
 dni INT(8),
 telefono INT(10)
 );
 
 create table Historial_Reservas(
 ID_historial INT NOT NULL primary key,
 cliente varchar (40),
 reserva datetime,
 foreign key (reserva) references reserva
 );
